1. Oracle Setup (Step-by-Step)
1.1 Login as SYSTEM
docker exec -it oracle-db sqlplus system/Oracle123@FREEPDB1
1.2 Create Users
-- create schema owner
CREATE USER user_db IDENTIFIED BY password;
GRANT CONNECT, RESOURCE TO user_db;

-- create app user
CREATE USER app_user IDENTIFIED BY app_pass;
GRANT CONNECT, RESOURCE TO app_user;
1.3 Create Table (as user_db)
ALTER SESSION SET CURRENT_SCHEMA = user_db;

CREATE TABLE users (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR2(100),
    email VARCHAR2(100),
    age NUMBER
);
1.4 Grant Access
GRANT INSERT, SELECT, UPDATE, DELETE ON user_db.users TO app_user;

-- optional (clean mapping)
CREATE SYNONYM app_user.users FOR user_db.users;
