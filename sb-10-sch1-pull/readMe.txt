1) Create Table
CREATE TABLE APP_DATA (ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,REQ_ID VARCHAR2(50) NOT NULL,NAME VARCHAR2(100),TYPE VARCHAR2(50),STATUS_FLAG CHAR(1) DEFAULT 'U',RESPONSE_MSG CLOB,RETRY_COUNT NUMBER DEFAULT 0,CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,UPDATED_AT TIMESTAMP);
2) Index
CREATE INDEX IDX_APP_DATA_FLAG ON APP_DATA(STATUS_FLAG);
3) Constraints
ALTER TABLE APP_DATA ADD CONSTRAINT CHK_STATUS_FLAG CHECK (STATUS_FLAG IN ('U','R','E','S'));
ALTER TABLE APP_DATA ADD CONSTRAINT CHK_TYPE CHECK (TYPE IN ('single','multi'));
4) PROC1 — Get Single Record
CREATE OR REPLACE PROCEDURE GET_SINGLE_RECORD_PROC (O_ID OUT NUMBER,O_REQ_ID OUT VARCHAR2,O_NAME OUT VARCHAR2,O_STATUS OUT CHAR) AS BEGIN SELECT ID,REQ_ID,NAME,STATUS_FLAG INTO O_ID,O_REQ_ID,O_NAME,O_STATUS FROM APP_DATA WHERE STATUS_FLAG IN ('U','R') AND ROWNUM=1; EXCEPTION WHEN NO_DATA_FOUND THEN O_ID:=NULL;O_REQ_ID:=NULL;O_NAME:=NULL;O_STATUS:=NULL; END;
/
5) PROC2 — Get Multiple Records (FINAL FIXED)
CREATE OR REPLACE PROCEDURE GET_MULTI_RECORD_PROC (P_LIMIT IN NUMBER,P_OFFSET IN NUMBER,O_CURSOR OUT SYS_REFCURSOR) AS BEGIN OPEN O_CURSOR FOR SELECT ID,REQ_ID,NAME,TYPE,STATUS_FLAG,RESPONSE_MSG,RETRY_COUNT,CREATED_AT,UPDATED_AT FROM APP_DATA WHERE STATUS_FLAG IN ('U','R') ORDER BY CREATED_AT OFFSET P_OFFSET ROWS FETCH NEXT P_LIMIT ROWS ONLY; END;
/

👉 No FOR UPDATE (prevents ORA-02014 and ORA-01002)

6) PROC3 — Update Record
CREATE OR REPLACE PROCEDURE UPDATE_RECORD_PROC (P_ID IN NUMBER,P_STATUS_FLAG IN CHAR,P_RESPONSE_MSG IN CLOB) AS BEGIN UPDATE APP_DATA SET STATUS_FLAG=P_STATUS_FLAG,RESPONSE_MSG=P_RESPONSE_MSG,RETRY_COUNT=CASE WHEN P_STATUS_FLAG='R' THEN RETRY_COUNT+1 ELSE RETRY_COUNT END,UPDATED_AT=SYSTIMESTAMP WHERE ID=P_ID AND STATUS_FLAG IN ('U','R'); COMMIT; END;
/
7) Verify Objects
SELECT OBJECT_NAME,OBJECT_TYPE,STATUS FROM USER_OBJECTS WHERE OBJECT_NAME IN ('GET_SINGLE_RECORD_PROC','GET_MULTI_RECORD_PROC','UPDATE_RECORD_PROC');
