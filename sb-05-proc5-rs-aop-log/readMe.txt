    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NAME VARCHAR2(100),
    DEPT VARCHAR2(50),
    SALARY NUMBER,
    CREATED_AT DATE DEFAULT SYSDATE
);

5) CREATE INDEX (IMPORTANT FOR PERFORMANCE)
CREATE INDEX IDX_EMP_DEPT ON EMPLOYEE(DEPT);
6) CREATE STORED PROCEDURE (CURSOR + ERROR HANDLING)
CREATE OR REPLACE PROCEDURE GET_EMP_BY_DEPT (
    P_DEPT   IN VARCHAR2,
    P_CURSOR OUT SYS_REFCURSOR
)
AS
    V_COUNT NUMBER;
BEGIN
    -- Check data existence
    SELECT COUNT(*) INTO V_COUNT
    FROM EMPLOYEE
    WHERE DEPT = P_DEPT;

    IF V_COUNT = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'NO DATA FOUND');
    END IF;

    -- Return full dataset (no limit here)
    OPEN P_CURSOR FOR
        SELECT ID, NAME, DEPT, SALARY, CREATED_AT
        FROM EMPLOYEE
        WHERE DEPT = P_DEPT;

EXCEPTION
    WHEN OTHERS THEN
        RAISE_APPLICATION_ERROR(-20002, 'ERROR: ' || SQLERRM);
END;
/
7) TEST PROCEDURE IN SQLPLUS

8) GRANT PERMISSIONS (IMPORTANT)

Now allow other users (like app_user) to access.

8.1 Login as SYSTEM
docker exec -it oracle-db sqlplus system/Oracle123@FREEPDB1
8.2 Grant Access
GRANT SELECT, INSERT, UPDATE, DELETE ON user_db.EMPLOYEE TO app_user;

GRANT EXECUTE ON user_db.GET_EMP_BY_DEPT TO app_user;
8.3 (OPTIONAL) SYNONYM FOR CLEAN ACCESS

Login as app_user:

docker exec -it oracle-db sqlplus app_user/app_pass@FREEPDB1
CREATE SYNONYM EMPLOYEE FOR user_db.EMPLOYEE;
CREATE SYNONYM GET_EMP_BY_DEPT FOR user_db.GET_EMP_BY_DEPT;

Now app_user can call directly:

SELECT * FROM EMPLOYEE;
EXEC GET_EMP_BY_DEPT('IT', :rc);
